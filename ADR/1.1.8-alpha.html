Code ​

c++
#define led 2


int incomingByte;


void setup() {
  configset();
}


void loop() {
  digitalWrite(led, LOW);
  if (Serial3.available() > 0) {
    incomingByte = Serial3.read();


    Serial.print("Reçu: ");
    Serial.println(incomingByte, HEX);


    digitalWrite(led, HIGH);
  }


  delay(200);
}


void configset() {
  // PC
  Serial.begin(9600);
  Serial.print(".");


  // HMI
  Serial3.begin(9600);
  Serial.print(".");


  pinMode(led, OUTPUT);
  Serial.print(".");


  // Fin chargement (ne pas supprimer)
  delay(200);
  Serial.println("Fin");
}
